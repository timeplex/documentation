77 void plexi_get_queue_handler(void *request, void *response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset) {
78  plexi_reply_content_len = 0;
79  unsigned int accept = -1;
80  REST.get_header_accept(request, &accept);
81  if(accept == -1 || accept == REST.type.APPLICATION_JSON) {
82  // Run through all the neighbors. Each neighbor has one queue. There are two extra for the EBs and the broadcast messages.
83  // The function tsch_queue_get_nbr_next is defined by George in tsch_queue.h/.c
84  int first_item = 1;
85  char buf[32];
86  struct tsch_neighbor *neighbor = NULL;
87  for(neighbor = (struct tsch_neighbor *)tsch_queue_get_nbr_next(NULL);
88  neighbor != NULL;
89  neighbor = (struct tsch_neighbor *)tsch_queue_get_nbr_next(neighbor)) {
90  linkaddr_t tna = neighbor->addr; // get the link layer address of neighbor
91  int txlength = tsch_queue_packet_count(&tna); // get the size of his queue
92  int success = plexi_linkaddr_to_eui64(buf, &tna); //convert his address to string
93  if(success) { // if the address was valid
94  if(first_item) {
95  CONTENT_PRINTF("{");
96  } else {
97  CONTENT_PRINTF(",");
98  }
99  first_item = 0;
100  CONTENT_PRINTF("\"%s\":%d", buf, txlength); // put to the output string the new JSON item
101  }
102  }
103  if(!first_item) { //if you found at least one queue
104  CONTENT_PRINTF("}");
105  REST.set_header_content_type(response, REST.type.APPLICATION_JSON);
106  REST.set_response_payload(response, (uint8_t *)plexi_reply_content, plexi_reply_content_len);
107  } else { // if no queues
108  coap_set_status_code(response, NOT_FOUND_4_04);
109  coap_set_payload(response, "No neighbor was found", 21);
110  return;
111  }
112  } else {
113  coap_set_status_code(response, NOT_ACCEPTABLE_4_06);
114  return;
115  }
116 }
