105 int plexi_json_find_field(struct jsonparse_state *js, char *field_buf, int field_buf_len) {
106  int state=jsonparse_next(js);
107  while(state) {
108  switch(state) {
109  case JSON_TYPE_PAIR_NAME:
110  jsonparse_copy_value(js, field_buf, field_buf_len);
111  /* Move to ":" */
112  jsonparse_next(js);
113  /* Move to value and return its type */
114  return jsonparse_next(js);
115  default:
116  return state;
117  }
118  state=jsonparse_next(js);
119  }
120  return 0;
121 }

123 /* Utility function. Converts na field (string containing the lower 64bit of the IPv6) to
124  * 64-bit MAC. */
125 int plexi_eui64_to_linkaddr(const char *na_inbuf, int bufsize, linkaddr_t *linkaddress) {
126  int i;
127  char next_end_char = ':';
128  const char *na_inbuf_end = na_inbuf + bufsize - 1;
129  char *end;
130  unsigned val;
131  for(i=0; i<4; i++) {
132  if(na_inbuf >= na_inbuf_end) {
133  return 0;
134  }
135  if(i == 3) {
136  next_end_char = '\0';
137  }
138  val = (unsigned)strtoul(na_inbuf, &end, 16);
139  /* Check conversion */
140  if(end != na_inbuf && *end == next_end_char && errno != ERANGE) {
141  linkaddress->u8[2*i] = val >> 8;
142  linkaddress->u8[2*i+1] = val;
143  na_inbuf = end+1;
144  } else {
145  return 0;
146  }
147  }
148  /* We consider only links with IEEE EUI-64 identifier */
149  linkaddress->u8[0] ^= 0x02;
150  return 1;
151 }

153 int plexi_linkaddr_to_eui64(char *buf, linkaddr_t *addr) {
154  char *pointer = buf;
155  unsigned int i;
156  for(i = 0; i < sizeof(linkaddr_t); i++) {
157  if(i > 1 && i!=3 && i!=4 && i!=7) {
158  *pointer = ':';
159  pointer++;
160  }
161  if(i==4){
162  continue;
163  }
164  if(i==0) {
165  sprintf(pointer, "%x", addr->u8[i]^0x02);
166  pointer++;
167  } else {
168  sprintf(pointer, "%02x", addr->u8[i]);
169  pointer+=2;
170  }
171  }
172  return strlen(buf);
173 }
