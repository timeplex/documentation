313 void aggregate_statistics(uint16_t id, uint8_t metric, plexi_stats_value_st value) {
314  if(value != -1) {
315  if(metric == RSSI) {
316  if(temp_aggregate_stats.rssi==(int)0xFFFFFFFFFFFFFFFF) { temp_aggregate_stats.rssi = value; }
317  else { temp_aggregate_stats.rssi += value; }
318  temp_aggregate_stats.rssi_counter++;
319  } else if(metric == LQI) {
320  if(temp_aggregate_stats.lqi==-1) { temp_aggregate_stats.lqi = value; }
321  else { temp_aggregate_stats.lqi += value; }
322  temp_aggregate_stats.lqi_counter++;
323  } else if(metric == ETX) {
324  if(temp_aggregate_stats.etx==-1) { temp_aggregate_stats.etx = (int)value; }
325  else { temp_aggregate_stats.etx += (int)value; }
326  temp_aggregate_stats.etx_counter++;
327  } else if(metric == PDR) {
328  if(temp_aggregate_stats.pdr==-1) { temp_aggregate_stats.pdr = value; }
329  else { temp_aggregate_stats.pdr += value; }
330  temp_aggregate_stats.pdr_counter++;
331  } else if(metric == ASN) {
332  if(temp_aggregate_stats.asn==-1 || value > temp_aggregate_stats.asn) { temp_aggregate_stats.asn = value; }
333  temp_aggregate_stats.asn_counter++;
334  }
335  }
336 }
