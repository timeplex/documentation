223 void plexi_post_slotframe_handler(void* request, void* response, uint8_t *buffer, uint16_t preferred_size, int32_t *offset) {
224  plexi_reply_content_len = 0;
225  int state;
226  int request_content_len;
227  int first_item = 1;
228  const uint8_t *request_content;
229 
230  char field_buf[32] = "";
231  int new_sf_count = 0; /* The number of newly added slotframes */
232  int ns = 0; /* number of slots */
233  int fd = 0;
234  /* Add new slotframe */
235 
236  unsigned int accept = -1;
237  REST.get_header_accept(request, &accept);
238  if(accept == -1 || accept == REST.type.APPLICATION_JSON) {
239 
240  request_content_len = REST.get_request_payload(request, &request_content);
241 
242  struct jsonparse_state js;
243  jsonparse_setup(&js, (const char *)request_content, request_content_len);
244 
245  /* Start creating response */
246  CONTENT_PRINTF("[");
247 
248  /* Parse json input */
249  while((state=plexi_json_find_field(&js, field_buf, sizeof(field_buf)))) {
250  switch(state) {
251  case '{': /* New element */
252  ns = 0;
253  fd = 0;
254  break;
255  case '}': { /* End of current element */
256  struct tsch_slotframe *slotframe = (struct tsch_slotframe *)tsch_schedule_get_slotframe_by_handle(fd);
257  if(!first_item) {
258  CONTENT_PRINTF(",");
259  }
260  first_item = 0;
261  if(slotframe || fd<0) {
262  printf("PLEXI:! could not add slotframe %u with length %u\n", fd, ns);
263  CONTENT_PRINTF("0");
264  } else {
265  if(tsch_schedule_add_slotframe(fd, ns)) {
266  new_sf_count++;
267  printf("PLEXI: added slotframe %u with length %u\n", fd, ns);
268  CONTENT_PRINTF("1");
269  } else {
270  CONTENT_PRINTF("0");
271  }
272  }
273  break;
274  }
275  case JSON_TYPE_NUMBER: //Try to remove the if statement and change { to [ on line 601.
276  if(!strncmp(field_buf, FRAME_ID_LABEL, sizeof(field_buf))) {
277  fd = jsonparse_get_value_as_int(&js);
278  } else if(!strncmp(field_buf, FRAME_SLOTS_LABEL, sizeof(field_buf))) {
279  ns = jsonparse_get_value_as_int(&js);
280  }
281  break;
282  }
283  }
284  CONTENT_PRINTF("]");
285  /* Check if json parsing succeeded */
286  if(js.error == JSON_ERROR_OK) {
287  REST.set_header_content_type(response, REST.type.APPLICATION_JSON);
288  REST.set_response_payload(response, (uint8_t *)plexi_reply_content, plexi_reply_content_len);
289  } else {
290  coap_set_status_code(response, BAD_REQUEST_4_00);
291  coap_set_payload(response, "Can only support JSON payload format", 36);
292  }
293  } else {
294  coap_set_status_code(response, NOT_ACCEPTABLE_4_06);
295  return;
296  }
