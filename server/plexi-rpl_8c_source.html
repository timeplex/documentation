71 static void plexi_get_dag_handler(void *request,
72  void *response,
73  uint8_t *buffer,
74  uint16_t preferred_size,
75  int32_t *offset) {
76  unsigned int accept = -1;
77  REST.get_header_accept(request, &accept);
78  /* make sure the request accepts JSON reply or does not specify the reply type */
79  if(accept == -1 || accept == REST.type.APPLICATION_JSON) {
80  plexi_reply_content_len = 0;
81  CONTENT_PRINTF("{");
82  // TODO: get details per dag id other than the default
83  /* Ask RPL to provide the preferred parent or if not known (e.g. LBR) leave it empty */
84  rpl_parent_t *parent = rpl_get_any_dag()->preferred_parent;
85  if(parent != NULL) {
86  /* retrieve the IP address of the parent */
87  uip_ipaddr_t *parent_addr = rpl_get_parent_ipaddr(parent);
88  CONTENT_PRINTF("\"%s\":[\"%x:%x:%x:%x\"]", DAG_PARENT_LABEL,
89  UIP_HTONS(parent_addr->u16[4]), UIP_HTONS(parent_addr->u16[5]),
90  UIP_HTONS(parent_addr->u16[6]), UIP_HTONS(parent_addr->u16[7])
91  );
92  } else {
93  CONTENT_PRINTF("\"%s\":[]", DAG_PARENT_LABEL);
94  }
95 
96  CONTENT_PRINTF(",\"%s\":[", DAG_CHILD_LABEL);
97 
98  nbr_table_item_t *r;
99  for(r = nbr_table_head(nbr_routes); r != NULL; r = nbr_table_next(nbr_routes, r))
100  {
101  if(r != nbr_table_head(nbr_routes))
102  {
103  CONTENT_PRINTF(",");
104  }
105  linkaddr_t *addr = (linkaddr_t *)nbr_table_get_lladdr(nbr_routes,r);
106  CONTENT_PRINTF("\"2%02x:%02x%02x:%02x:%02x%02x\"",
107  UIP_HTONS(addr->u8[1]), UIP_HTONS(addr->u8[2]), UIP_HTONS(addr->u8[3]),
108  UIP_HTONS(addr->u8[5]), UIP_HTONS(addr->u8[6]), UIP_HTONS(addr->u8[7])
109  );
110  }
111  CONTENT_PRINTF("]}");
112  /* Build the header of the reply */
113  REST.set_header_content_type(response, REST.type.APPLICATION_JSON);
114  /* Build the payload of the reply */
115  REST.set_response_payload(response, (uint8_t *)plexi_reply_content, plexi_reply_content_len);
